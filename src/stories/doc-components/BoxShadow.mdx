import { Canvas, SourceState } from '@storybook/addon-docs';
import Colors from '../../scss/riseup-js-colors.module.scss';
import { BoxShadow } from '@/tokens/box-shadows';
import { BorderRadius } from '@/tokens/border-radius';
import { SpacingContainer } from './UtilComponents.mdx';
import { CopyToClipboard } from './CopyToClipboard.mdx';

export const BoxShadows = () => {
  return <Canvas withSource={SourceState.NONE}>
    <SpacingContainer columnGap={48}>
      {
        Object.entries(BoxShadow).map(([name, definition]) => {
          const boxStyle = {
            height: 150,
            width: 150,
            border: `1px solid ${Colors.riseupGray0}`,
          boxShadow: definition,
            borderRadius: BorderRadius.medium,
          };
          return <div className={'ri-small-body flex-column align-center'}>
            <div style={boxStyle} />
            <div className={'margin-top-xxs'}>{definition}</div>
            <div className={'ri-bold-label'}>{name.toUpperCase()}</div>
            <CopyToClipboard text={`.box-shadow-${name.toLowerCase()}`} copiedText={`box-shadow-${name.toLowerCase()}`}/>
          </div>;
        })
      }
    </SpacingContainer>
  </Canvas>;
};
